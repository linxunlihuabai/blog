# 二叉树的遍历

二叉树的遍历主要分为深度优先遍历和广度优先遍历

## 深度优先遍历

1. 先序遍历（根左右）
2. 中序遍历（左根右）
3. 后序遍历（左右根）

### 先序遍历

递归实现

```js
function ergodic(root){
    if(!root){
        return;
    }

    console.log(root.val);
    ergodic(root.left);
    ergodic(root.right);
}
```

迭代实现

```js
function ergodic(root){
    if(!root){
        return;
    }

    let stack = [root];

    while(stack.length){
        let top = stack.pop();
        console.log(top.val);

        if(top.right){
            stack.push(top.right)
        }

        if(top.left){

            stack.push(top.left)
        }
    }
}
```

### 中序遍历

递归实现

```js
function ergodic(root){
    if(!root){
        return;
    }

    ergodic(root.left);
    console.log(root.val);
    ergodic(root.right);
}
```

迭代实现

```js
function ergodic(root){
    if(!root){
        return;
    }

    let stack = [];

    while(root || stack.length){
        while(root){
            stack.push(root);
            root = root.left;
        }

        let top = stack.pop();
        console.log(top.val);
        root = top.right;
    }
}
```

### 后序遍历

递归实现

```js
function ergodic(root){
    if(!root){
        return;
    }

    ergodic(root.left);
    ergodic(root.right);
    console.log(root.val);
}
```

迭代实现

```js
function ergodic(root){
    if(!root){
        return;
    }

    let stack1 = [root];
    let stack2 = [];

    while(stack1.length){
        let top = stack1.pop();
        stack2.push(top);
        if(top.left){
            stack1.push(top.left);
        }
        if(top.right){
            stack1.push(top.right);
        }
    }

    while(stack2.length){
        console.log(stack2.pop().val);
    }
}
```
